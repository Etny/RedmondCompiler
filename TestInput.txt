using System;
using System.Collections.Generic;

namespace BaseTest
{
    class Program
    {

        static void Main()
        {
            PrintBase((Derived) GetBase());
            new Derived().Print();
        }

        static Base GetBase()
            => new Derived();

        static int Num()
            => 2;

        static void PrintBase(Derived b)
            => b.Print();
    }

    class Base
    {
        public virtual void Print()
            => Console.WriteLine("Base Print");
    }

    class Derived : Base
    {
        public override void Print()
            => Console.WriteLine("Derived Print");
    }
}
