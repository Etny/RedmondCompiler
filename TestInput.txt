using System;
using System.Collections.Generic;

namespace BaseTest
{
    class Program
    {
        static void Main()
        {
            Generic<int, string> g = new Generic<int, string>(2);
            Console.WriteLine(g.Get("Lol"));
        }
    }

    class Generic<T, U>
    {
        T Thing;

        public Generic(T i)
        {
            Thing = i;
        }
        
        public T Get(U u) 
        {
            Console.WriteLine(u);
            return Thing;
        }
    }
}
