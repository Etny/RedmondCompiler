using System;
using System.Collections.Generic;

namespace BaseTest
{
    class Program
    {

        static void Main()
        {
            Derived t = new Derived();
            printThing(t);
        }

        static void printThing(Thing t)
        {
            Console.WriteLine(t.Property);
        }
    }

    class Thing
    {
        public virtual int Property => 2;
    }

    class Derived : Thing
    {
        public override int Property => 5;
    }
}
